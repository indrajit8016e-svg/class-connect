# ==============================================================================
# ClassConnect - Production Environment Requirements (Vercel & Render.com)
# Version: 4.0
# ==============================================================================

# 1. Hosting Platforms
# ------------------------------------------------------------------------------
# Frontend: Vercel (Optimized for Vite + React)
# Backend: Render.com (Web Service for Node.js API)
# Database: Render PostgreSQL (Managed Instance)

# 2. Runtime Environment
# ------------------------------------------------------------------------------
node >= 18.20.0 (Render Default)
npm >= 10.0.0
typescript >= 5.0.0

# 3. Database (Primary Persistence)
# ------------------------------------------------------------------------------
postgresql >= 14.0 (Render Managed Postgres)
# Required Env: DATABASE_URL

# 4. Security & Anti-Virus
# ------------------------------------------------------------------------------
# Note: Render does not include ClamAV by default.
# The app includes a fallback for development. For production:
# 1. Use a custom Dockerfile with ClamAV installed.
# 2. Or rely on the application's verification status logic.

# 5. Storage (Infrastructure)
# ------------------------------------------------------------------------------
# Primary: Cloudinary (Persistent Cloud Storage)
# Required Envs: CLOUDINARY_CLOUD_NAME, CLOUDINARY_API_KEY, CLOUDINARY_API_SECRET
# Temporary: Render Local File System (/server/uploads - Ephemeral)

# 6. Real-time Infrastructure
# ------------------------------------------------------------------------------
# Default: In-memory Socket.IO adapter (Single Instance)
# Scaling: Redis (Render Redis) required if horizontal scaling is enabled.

# 7. Backend Stack (Summary)
# ------------------------------------------------------------------------------
express          # Web Framework
socket.io        # Real-time Engine
pg               # PostgreSQL Client
bcryptjs         # Password Hashing
jsonwebtoken     # Authentication
multer           # Multipart/File Upload
cloudinary       # Cloud Storage Service
clamscan         # ClamAV Node Interface (with fallback)
dotenv           # Env Management
cors             # Security Headers

# 8. Frontend Stack (Summary)
# ------------------------------------------------------------------------------
react >= 18.0
vite             # Build Tool (Vercel Build Command: npm run build)
tailwindcss      # Styling
lucide-react     # Iconography
axios            # API Client (VITE_API_URL required)
socket.io-client # Real-time Client (VITE_SOCKET_URL required)
@tanstack/react-query # State/Data Fetching

# 9. Deployment Configuration
# ------------------------------------------------------------------------------
# vercel.json: Handles SPA client-side routing on Vercel.
# server/package.json: Build script copies SQL schemas for Render DB init.
